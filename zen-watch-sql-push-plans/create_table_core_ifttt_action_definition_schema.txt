CREATE TABLE ifttt_action_definition (
  id INT NOT NULL AUTO_INCREMENT,
  dev_id INT NOT NULL,
  categ_id INT NOT NULL,
  target_resource_name VARCHAR(500) NOT NULL,
  action_name VARCHAR(255) NOT NULL,
  action_description VARCHAR(5000) NOT NULL,
  action_signature TEXT NOT NULL,
  action_code TEXT NOT NULL,
  action_metadata JSON,
  action_expected_output TEXT NOT NULL,
  action_expected_output_description TEXT NOT NULL,
  created_ts TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_ts TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY(id),
  INDEX IADTRN (target_resource_name ASC),
  INDEX IADAN (action_name ASC),
  INDEX IADCTS (created_ts ASC),
  INDEX IADUTS (updated_ts ASC),
  CONSTRAINT IADD_FK FOREIGN KEY(dev_id) REFERENCES developer(id) ON DELETE RESTRICT ON UPDATE CASCADE,
  CONSTRAINT IADCID_FK FOREIGN KEY(categ_id) REFERENCES ifttt_category(id) ON DELETE RESTRICT ON UPDATE CASCADE
);
