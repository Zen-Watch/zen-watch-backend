CREATE TABLE ifttt_trigger_run_history (
  id INT NOT NULL AUTO_INCREMENT,
  ifttt_trigger_run_history_worker_shard_id INT NOT NULL,
  dev_id INT NOT NULL,
  is_trigger_trusted_source TINYINT(1) DEFAULT 0,
  is_trigger_compute_intensive TINYINT(1) DEFAULT 0,
  is_trigger_push_mechanism TINYINT(1) DEFAULT 0,
  trigger_target_resource_name VARCHAR(500) NOT NULL,
  ifttt_instance_id INT NOT NULL,
  trigger_run_status INT NOT NULL,
  trigger_run_info JSON NOT NULL,
  trigger_run_output JSON,
  created_ts TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_ts TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY(id),
  INDEX ITRH_ITRHWSI (ifttt_trigger_run_history_worker_shard_id ASC),
  INDEX ITRHITTS (is_trigger_trusted_source),
  INDEX ITRHITCI (is_trigger_compute_intensive),
  INDEX ITRHITPM (is_trigger_push_mechanism),
  INDEX ITRHTTRN (trigger_target_resource_name ASC),
  INDEX ITRHTRS (trigger_run_status ASC),
  INDEX ITRHCTS (created_ts ASC),
  INDEX ITRHUTS (updated_ts ASC),
  CONSTRAINT ITRHD_FK FOREIGN KEY(dev_id) REFERENCES developer(id) ON DELETE RESTRICT ON UPDATE CASCADE,
  CONSTRAINT ITRHIID_FK FOREIGN KEY(ifttt_instance_id) REFERENCES ifttt_instance(id) ON DELETE RESTRICT ON UPDATE CASCADE
);
