CREATE TABLE ifttt_definition (
  id INT NOT NULL AUTO_INCREMENT,
  ifttt_worker_shard_id INT NOT NULL,
  dev_id INT NOT NULL,
  name VARCHAR(255) NOT NULL,
  trigger_info TEXT NOT NULL,
  actions_info TEXT NOT NULL,
  created_ts TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_ts TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY(id),
  INDEX IFTD_IWSI (ifttt_worker_shard_id ASC),
  INDEX IFTDN (name ASC),
  INDEX IFTDCTS (created_ts ASC),
  INDEX IFTDUTS (updated_ts ASC),
  CONSTRAINT IFTDD_FK FOREIGN KEY(dev_id) REFERENCES developer(id) ON DELETE RESTRICT ON UPDATE CASCADE
);
